#include <iostream>
using namespace std;
template <typename T>
void swap(T &a, T& b)
{
    T t = a;
    a = b;
    b = t;
}
template <typename T>
class Stack
{
public:
    Stack(int size)
    {
        space = new T[size];
        top = 0;
    }
    ~Stack()
    {
        delete[] space;
    }
    T pop()
    {
            return space[top--];     
    }
    T top()
    {
            return space[top - 1];
    }
    void push(T a)
    {
        space[top++] = a;
    }
    bool isempty()
    {
        return top == 1024;
    }
    T* space;
    int top;
};
